tworzenie bazy:

-- Tabela dla krajów
CREATE TABLE countries (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL
);

-- Tabela dla imion
CREATE TABLE names (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    country_id INT,
    FOREIGN KEY (country_id) REFERENCES countries(id)
);

-- Tabela dla nazwisk
CREATE TABLE surnames (
    id INT AUTO_INCREMENT PRIMARY KEY,
    surname VARCHAR(255) NOT NULL,
    country_id INT,
    FOREIGN KEY (country_id) REFERENCES countries(id)
);

-- Tabela dla cech osobowości
CREATE TABLE traits (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    description TEXT NOT NULL
);

-- Tabela dla featów
CREATE TABLE feats (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    description TEXT NOT NULL,
    skill_modifiers JSON NOT NULL
);

-- Tabela dla gotowych NPC
CREATE TABLE npc (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name_id INT,
    surname_id INT,
    country_id INT,
    brawn INT NOT NULL,
    agility INT NOT NULL,
    intellect INT NOT NULL,
    cunning INT NOT NULL,
    willpower INT NOT NULL,
    presence INT NOT NULL,
    poczytalnosc INT NOT NULL,
    szczescie INT NOT NULL,
    atletyka INT NOT NULL,
    odpornosc INT NOT NULL,
    bijatyka INT NOT NULL,
    bron_dwureczna INT NOT NULL,
    lucznictwo INT NOT NULL,
    machlojki INT NOT NULL,
    koordynacja INT NOT NULL,
    bron_lekka INT NOT NULL,
    rewolwery INT NOT NULL,
    rzucanie INT NOT NULL,
    karabiny INT NOT NULL,
    mity INT NOT NULL,
    astronawigacja INT NOT NULL,
    geografia INT NOT NULL,
    wycena INT NOT NULL,
    natura INT NOT NULL,
    alchemia INT NOT NULL,
    medycyna INT NOT NULL,
    artylery INT NOT NULL,
    pierwsza_pomoc INT NOT NULL,
    pismiennosc INT NOT NULL,
    ukrywanie INT NOT NULL,
    orientacja INT NOT NULL,
    przeszukiwanie INT NOT NULL,
    oszustwo INT NOT NULL,
    czujnosc INT NOT NULL,
    opanowanie INT NOT NULL,
    przymuszenie INT NOT NULL,
    urok INT NOT NULL,
    dowodzenie INT NOT NULL,
    negocjacje INT NOT NULL,
    jezyki_obce INT NOT NULL,
    traits JSON NOT NULL,
    feats JSON NOT NULL,
    FOREIGN KEY (name_id) REFERENCES names(id),
    FOREIGN KEY (surname_id) REFERENCES surnames(id),
    FOREIGN KEY (country_id) REFERENCES countries(id)
);
